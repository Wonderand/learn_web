        function fetchAndSetPDFThumbnail() {
            const userValue = reader.userValue;
            // if (!userValue) return;
            const formData = new FormData();
            formData.append('fl', reader.fileValue);
            var url = getPDFThumbnailAsStream + "?fl=" + reader.fileValue;
            const eventSource = new EventSource(url);
            // 监听SSE事件
            var elementById = document.getElementById('thumbnailView1');
            elementById.style.height = '78%'
        
            eventSource.onmessage = function (event) {
                // console.log('Received message:', event.data);
                const data = JSON.parse(event.data);
                // 检查是否有错误
                if (data.error) {
                    console.error("Error: " + data.error);
                } else {
                    // console.log(data)
                    var a = document.createElement('a');
                    a.href = '#page=' + (data.page);
                    // 创建一个新的元素
                    var img = document.createElement('img');
        
                    // 设置img的src为Base64编码的图片数据
                    img.src = data.url;
        
                    // 设置图片的样式
                    img.style.width = '98px'; // 注意：原要求为98px而非100px
                    img.style.height = '138px'; // 同上，要求为138px而非140px
                    img.style.border = '1px solid #000'; // 添加边框样式
        
                    // 为img元素添加必要的属性，以便于管理和识别
                    img.setAttribute('aria-label', '页面 ' + data.page + ' 的缩略图');
                    img.classList.add('thumbnailImage'); // 假设 thumbnailImage 是一个CSS类名
        
                    // 查找或创建对应的缩略图容器，并将img元素插入其中
                    // 注意：原始HTML结构中并没有直接给出每个缩略图的唯一容器标识，这里假设每个缩略图直接附加到elementById下
                    var thumbnailContainer = document.createElement('div');
                    // thumbnailContainer.classList.add('thumbnail', 'selected');
                    thumbnailContainer.classList.add('thumbnail'); // 假设.selected也是一个CSS类名
                    thumbnailContainer.id = 'page' + data.page;
                    thumbnailContainer.setAttribute('data-page-number', data.page);
                    thumbnailContainer.setAttribute('data-loaded', 'true');
        
                    // 创建缩略图的选择环元素
                    var selectionRing = document.createElement('div');
                    selectionRing.classList.add('thumbnailSelectionRing');
                    selectionRing.style.width = '100px'; // 或者根据需要调整
                    selectionRing.style.height = '140px'; // 同上
        
                    // 构建完整的结构：selectionRing > img
                    selectionRing.appendChild(img);
                    thumbnailContainer.appendChild(selectionRing);
                    a.appendChild(thumbnailContainer)
                    elementById.appendChild(a);
                    // thumbnailContainer.addEventListener('click', function (){
                    //     thumbnailContainer.classList.add("selected");
                    // })
                }
            };
            // 处理连接打开事件
            eventSource.onopen = function () {
                console.log('SSE connection opened');
            };
        
            // 处理连接关闭事件
            eventSource.onclose = function () {
                console.log('SSE connection closed');
            };
        
            // 处理错误事件
            eventSource.onerror = function (error) {
                console.error('SSE error:', error);
                eventSource.close();
            };
        }
